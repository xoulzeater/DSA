
CREATE TABLE "RESOURCE" (
	"RES_ID" INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	"NAME" VARCHAR(30),
	"TYPE" VARCHAR(30),
	"QUANTITY" INTEGER,
	"STATUS" VARCHAR(20)
);

CREATE TABLE "VICTIM" (
	"VICTIM_ID" INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	"NAME" VARCHAR(30),
	"DOB" DATE,
	"ADDRESS" VARCHAR(100),
	"STATUS" VARCHAR(20),
	"GENDER" VARCHAR(20)
	
);


CREATE TABLE "RESOURCEASSIGN" (
	"RES_ID" INTEGER,
	"VICTIM_ID" INTEGER,
	"ASSIGN_TIME" DATE,
	primary key ("RES_ID","VICTIM_ID"),
	foreign key ("RES_ID") references "RESOURCE"("RES_ID"),
	foreign key ("VICTIM_ID") references "VICTIM"("VICTIM_ID")
);

CREATE TABLE "TASK" (
	"TASK_ID" INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	"NAME" VARCHAR(30),
	"DESCRIPTION" VARCHAR(200),
	"MANPOWER" INTEGER,
	"START_TIME" DATE,
	"END_TIME" DATE	
);

CREATE TABLE "STAFF" (
	"STAFF_ID" INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	"NAME" VARCHAR(30),
	"DOB" DATE,
	"STATUS" VARCHAR(30),
	"GENDER" VARCHAR(15)
);

CREATE TABLE "TASKASSIGN" (
	"TASK_ID" INTEGER,
	"STAFF_ID" INTEGER,
	"TIME" DATE,
	PRIMARY KEY ("TASK_ID","STAFF_ID")
);

CREATE TABLE "MEDICAL" (
	"MED_ID" INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
	"TIME" TIME,
	"VICTIM_ID" INTEGER,
	"STAFF_ID" INTEGER,
	"DESCRIPTION" VARCHAR(200)
);
